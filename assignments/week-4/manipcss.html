Manipulating CSS Class Properties Using JavaScript.

Overview

When it comes to Javascript we do not want to manipulate the CSS directly. It is best to have CSS set up that can do the manipulation. This is helpful when collaborating as a team. One team member can focus on CSS while the other can deal with the JavaScript.

The two ways to manipulate classes in plain Javascript are with className and classList. Using className is a more direct approach and will work in every browser, whereas using classList is much simpler, but may not be supported in every browser. We will focus our efforts here on classList

Here are some example methods of the classList API.

  • add(String [,String])
  • remove(String [,String])
  • item(Number)
  • toggle(String [,force])
  • contains(String)
  • replace(oldClass, newClass)

                    // Class Manipulation
                    // manipulate.js
                    // This function will hide 
                function hide(){
                var divElement = document.querySelector('#manipulate');
                if (divElement.classList.contains('unhide') === true) {
                divElement.classList.add('hide');
                divElement.classList.remove('unhide');
                }
                }
                    // This function will unhide 
                function unhide(){
                var divElement = document.querySelector("#manipulate");
                if (divElement.classList.contains('hide') === true) {
                divElement.classList.replace('hide', 'unhide');
                }
                }
                    // This function will color the text red
                function redText(){
                var divElement = document.querySelector("#manipulate");
                divElement.classList.add('redText');
                }
                    // This function will return the color to black
                function blackText(){
                var divElement = document.querySelector("#manipulate");
                divElement.classList.remove('redText');
                }

                    // this funtion will toggle between shrink and original size
                function shrink(){
                var divElement = document.querySelector("#manipulate");
                divElement.classList.toggle('shrink');
                }
